```bash

     -H 'X-Dao-Auth: <auth_token>' \
     -H 'Content-Type: application/json' \
     -d '{"model":"sonnet4.5","messages":[{"role":"user","content":"道可道"}]}' \
     https://omni-router-app-tunnel-<id>.devinapps.com/port/7780/v1/chat/completions
```

```python
from openai import OpenAI
client = OpenAI(
    base_url="https://user:<basic_pass>@omni-router-app-tunnel-<id>.devinapps.com/port/7780/v1",

    model="sonnet4.5",  # 或 opus / gpt5 / gemini / kimi / 任 MODEL_xxx · 共 228 件可选
```
