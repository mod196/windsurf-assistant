### §5 · F 层 (主进程退) 之实

```js
// 14a 试 VSCode 标命
await vscode.commands.executeCommand("workbench.action.quit");

// 14b taskkill fallback (Windows 终极手段)
const wmicOut = cp.execSync(
  'wmic process where "Name=\'Windsurf.exe\'" get ProcessId,CommandLine /format:csv',
  { encoding: "utf8", windowsHide: true }
);
// 找主进程 PID (CommandLine 不含 --type=)
for (const line of wmicOut.split("\n")) {
  const fields = line.split(",");
  const cmd = fields[1] || "";
  const pid = (fields[fields.length - 1] || "").trim();
  if (cmd && !/--type=/.test(cmd) && /^\d+$/.test(pid)) {
    mainPid = pid;
    break;
  }
}
if (mainPid) {
  cp.execSync(`taskkill /F /PID ${mainPid}`, { windowsHide: true });
  // ★ 此后 extension 死 · zhou Windsurf 整个退出 · 一切内存清空
}
```
